## 4. Hoe deze inventaris updaten

```bash
# Quizvragen per vak/level
node --input-type=module -e "
import('./src/constants.js').then(m => {
  const sq = m.SAMPLE_QUESTIONS;
  const out = {};
  for (const s of Object.keys(sq)) {
    out[s] = {};
    for (const l of Object.keys(sq[s])) out[s][l] = Array.isArray(sq[s][l]) ? sq[s][l].length : '?';
  }
  console.log(JSON.stringify(out, null, 2));
});"

# Leerpaden metadata
node -e "
const fs = require('fs');
const dir = 'src/learnPaths';
const files = fs.readdirSync(dir).filter(f => /\.(js|jsx)$/.test(f) && !/(index|test|generated|subjectMapping)/.test(f));
const rows = files.map(f => {
  const t = fs.readFileSync(dir+'/'+f, 'utf8');
  const g = re => { const m = t.match(re); return m ? m[1] : ''; };
  return { id: g(/\bid:\s*['\"]([^'\"]+)['\"]/), subject: g(/\bsubject:\s*['\"]([^'\"]+)['\"]/), level: g(/\blevel:\s*['\"]([^'\"]+)['\"]/), title: g(/\btitle:\s*['\"]([^'\"]+)['\"]/) };
});
console.log(JSON.stringify(rows.sort((a,b) => (a.subject||'').localeCompare(b.subject||'')), null, 2));"
```
